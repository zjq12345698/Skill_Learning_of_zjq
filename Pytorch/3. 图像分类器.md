### 定义神经网络
该神经网络为3通道的图片

```python
import torch.nn as nn
import torch.nn.functional as F

class Net(nn.Module):
    def __init__(self):
        super(Net, self).__init__()
        # 输入通道数为3，输出通道数为6，卷积核大小为5x5
        self.conv1 = nn.Conv2d(3, 6, 5)
        # 池化核大小为2x2，步长为2
        self.pool = nn.MaxPool2d(2, 2)
        # 输入通道数为6，输出通道数为16，卷积核大小为5x5
        self.conv2 = nn.Conv2d(6, 16, 5)
        # 全连接层，输入维度为16*5*5，输出维度为120
        self.fc1 = nn.Linear(16 * 5 * 5, 120)
        # 全连接层，输入维度为120，输出维度为84
        self.fc2 = nn.Linear(120, 84)
        # 全连接层，输入维度为84，输出维度为10
        self.fc3 = nn.Linear(84, 10)

    def forward(self, x):
        # 输入x经过卷积，ReLU激活，池化操作
        x = self.pool(F.relu(self.conv1(x)))
        # 再经过一次卷积，ReLU激活，池化操作
        x = self.pool(F.relu(self.conv2(x)))
        # 将张量展开为一维向量
        x = x.view(-1, 16 * 5 * 5)
        # 经过第一个全连接层，ReLU激活
        x = F.relu(self.fc1(x))
        # 经过第二个全连接层，ReLU激活
        x = F.relu(self.fc2(x))
        # 经过第三个全连接层，输出预测结果
        x = self.fc3(x)
        return x

# 创建网络实例
net = Net()

```

### 定义一个损失函数和优化器

```python
criterion = nn.CrossEntropyLoss()
```

### 训练网络
在数据迭代器上循环传给网络和优化器

```python
for epoch in range(2):  # 多次循环遍历数据集

    running_loss = 0.0
    for i, data in enumerate(trainloader, 0):
        # 获取输入数据
        inputs, labels = data

        # 将梯度缓存清零
        optimizer.zero_grad()

        # 前向传播 + 反向传播 + 优化
        outputs = net(inputs)  # 输入经过神经网络得到输出
        loss = criterion(outputs, labels)  # 计算损失函数
        loss.backward()  # 反向传播，计算梯度
        optimizer.step()  # 优化网络参数

        # 打印统计信息
        running_loss += loss.item()
        if i % 2000 == 1999:  # 每2000个小批次打印一次
            print('[%d, %5d] loss: %.3f' %
                  (epoch + 1, i + 1, running_loss / 2000))
            running_loss = 0.0

print('Finished Training')  # 结束训练
```
